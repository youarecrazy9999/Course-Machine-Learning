from matplotlib.colors import ListedColormap
import matplotlib.pyplot as plt

def plot_decision_regions(X, y, classifier, test_idx = None, resolution = 0.02):
    # setup marker generator and color map
    markers = ('s', 'x', 'o', '^', 'v')
    colors = ('red', 'blue', 'lightgreen', 'gray', 'cyan')
    cmap = ListedColormap(colors[:len(np.unique(y))])

    # plot the decision surface
    x1_min, x1_max = X[:, 0].min() - 1, X[:, 0].max() + 1
    x2_min, x2_max = X[:, 1].min() - 1, X[:, 1].max() + 1
    xx1, xx2 = np.meshgrid(np.arange(x1_min, x1_max, resolution),
                           np.arange(x2_min, x2_max, resolution))
    Z = classifier.predict(np.array([xx1.ravel(), xx2.ravel()]).T)
    Z = Z.reshape(xx1.shape)
    plt.contourf(xx1, xx2, Z, alpha=0.3, cmap=cmap)
    plt.xlim(xx1.min(), xx1.max())
    plt.ylim(xx2.min(), xx2.max())

    for idx, cl in enumerate(np.unique(y)):
        plt.scatter(x=X[y == cl, 0], 
                    y=X[y == cl, 1],
                    alpha=0.8, 
                    c=colors[idx],
                    marker=markers[idx], 
                    label=cl, 
                    edgecolor='black')

    # highlight test examples
    if test_idx:
        # plot all examples
        X_test, y_test = X[test_idx, :], y[test_idx]

        plt.scatter(X_test[:, 0],
                    X_test[:, 1],
                    c='',
                    edgecolor='black',
                    alpha=1.0,
                    linewidth=1,
                    marker='o',
                    s=100, 
                    label='test set')

TỈ SỐ ODD¶

Gi p l xc sut xy ra ca bin c p => ODD1 = p Vy xc sut khng xy ra cc bin c p l 1-p => ODD2 = 1-p

Ta c t s Odd Ratio ( tng quan gia 2 bin c hay c th ni l bin c p c li hay khng)

Do xc sut p xy ra c gi tr trong khong (0,1) nn ta c th log-odds, nh ngha l logit(p), ph hp hn cho cc thut ton hi quy tuyn tnh.

$OR:$ $\frac{ODD1}{ODD2}$ = $\frac{p}{1-p}$

$logit(p) = log\frac{p}{1-p}$

Hm logit nhn gi tr u vo nm trong khong (0,1) v bin i ton b gi tr trn ton b phm vi s thc ($-\infty$,$+\infty$), khi ta c th s dng kt hp quan h tuyn tnh gia cc feature z = $(w^Tx)$ v xc sut xy ra p:

$logit(p(y=1|x)) = w_0x_0 + w_1x_1 + ... + w_mx_m = \sum{w_i}{x_i} = w^Tx $

$ln\frac{p}{1-p} = z$

$\Leftrightarrow \frac{p}{1-p} = e^z$

$\Leftrightarrow p = e^z(1-p) = e^z - e^zp$

$\Leftrightarrow p = \frac{e^z}{1+e^z} = \frac{1}{1+e^{-z}}$

Nh vy ta t $\phi(z) = \frac{1}{1+e^{-z}}$ l activation function ca Logistic Regression. Hm $\phi(z)$ c dng sigmoid nn cn gi l (logistic) sigmoid function.

LOSS FUNCTION¶

Nu nh MSE l phng php c s dng c tnh cc tham s trong Adaline th trong Logistic Regression, ta dng maximum likelihood estimate c tnh tham s $w$.

Ta xt N mu d liu c gn nhn 0 v 1:(coi 2 nhn l 2 bin c)

Ta thy c, xc sut 1 im d liu c gn nhn $y=1$ l $p(y=1|x_i;w) = \phi(z) = \phi(w^Tx_i)$

Tng t, xc sut 1 im d liu gn nhn $y=0$ l $p(y=0|x_i;w) = 1 - \phi(w^Tx_i)$

Vi 1 im d liu $z_i = \phi(w^Tx_i)$, ta c th vit chung 2 biu thc trn di dng:

$p(y_i|x_i;w) = z_i^{y_i}(1 - z_i)^{1-y_i}$

Mc ch l xc nh h s $w$ sao cho cc im d liu ng vi tng bin c nht c th. V th ta cn miximize tt c xc sut ca cc mu sao cho ln nht c th. Bi ton tr thnh maximum likelihood estimation.

Vi 1 tp N mu d liu v 2 nhn 0 v 1, ta c xc sut c iu kin tng qut: $P(y_i|X;w) = \prod_{i \in \{y=1\}}\phi(z_i) x \prod_{i \in \{y=0\}}(1 - \phi(z_i)) = \prod_{i=1}^{N}z_i^{y_i}(1 - z_i)^{1-y_i}$

t $P(y_i|X;w) = L(w)$, $L(w) = \prod_{i=1}^{N}z_i^{y_i}(1 - z_i)^{1-y_i}$ gi l likelihood function.

Ta cng c th n gin ha bng cch ly log hm $L(w)$.

$l(w) = logL(w) = \sum_{i=1}^{N}[{y_i}{log(\phi(z_i))} + (1 - y_i)log(1 - \phi(z_i))]$

Hm $l(w)$ gi l log-likelihood function.

S dng log-likelihood function :

Ti u trnh trng hp xy ra li trong tnh ton (numberial error) nu c $N$ ln, tch tr nn qu nh.

Bin php $*$ thnh php $+$, trnh vic s qu nh v thun tin o hm.

Chuyn thnh loss function, ta i du v dng Gradient Descent minimize hm l(w)

$J(w) = -l(w) = - \sum_{i=1}^{N}[{y_i}{log(\phi(z_i))} + (1 - y_i)log(1 - \phi(z_i))]$

hoc minimize dng cost function tng t Adaline (SSE): $J(w) = \sum_i \frac{1}{2}(\phi(z_i) - y_i)^2$

TỐI ƯU HÀM MẤT MÁT¶

Ta c loss function ti 1 im d liu $(x,y)$:

$j(\phi(z),y;w) = -ylog(\phi(z)) - (1 - y)log(1 - \phi(z))$

Nhn vo cng thc, ta c th thy: $$j(\phi(z),y;w) = \begin{cases} -log(\phi(z)) & if &y =1 \\ -log(1 - \phi(z)) & if &y =0 \end{cases}$$

o hm ring chi ph theo w:

$\nabla{j(w)} = \frac{\partial}{\partial{w}}j(w) = -(ylog(\phi(z)) + (1 - y)log(1 - \phi(z))\frac{\partial}{\partial{w}} $

$ = - (y\frac{1}{\phi(z)} - (1 - y)\frac{1}{1-\phi(z)})\frac{\partial}{\partial{w}}\phi(z)$ (1)

Ta c: $\frac{\partial}{\partial{w}} = \frac{\partial}{\partial{w}}\frac{1}{1+e^{-z}} = \phi(z)(1 - \phi(z))\frac{\partial}{\partial{w}}z = \phi(z)(1 - \phi(z))x $ (v o hm z l x) (2)

Vi (1) v (2), ta c:

$\frac{\partial}{\partial{w}}j(w) = -(y\frac{1}{\phi(z)} - (1 - y)\frac{1}{1-\phi(z)}) \phi(z) (1 - \phi(z)) x $

$= -(y - \phi(z))x$

Da vo Gradient Descent, ta c th update 1 trng s $w$ nh sau: $w := w + \eta \sum_{i=1}^{N}(y_i - \phi(z_i)){x}_i $

$\Rightarrow w := w + \Delta{w} $ vi $ \Delta{w} = \eta \nabla{j(w)} = -\eta \nabla{l(w)} $

Nh vy, vi ton dataset, trng s $W$ s c update: $W = W + \Delta{W}$ vi cost function $J(W)$.

class LogisticRegressionGD(object):
    '''Logistic Regression Classifier using gradient descent.
  
  Parameter
  ---------------
  eta : float
    Learning rate(between 0.0 and 1.0)
  n_iter : int
    Passes over the trainning dataaset.
  random_state : int
    Random number generator seed for random weight initialization.
    
  Attributes
  ---------------
  w_ : 1d-array
    Weights after fitting.
  cost_ : list
    Logistic cost function value in each epoch.
  '''
    def __init__(self, eta = 0.5, n_iter = 100, random_state =1):
        self.eta = eta
        self.n_iter = n_iter
        self.random_state = random_state
  
    def fit(self, X,y):
        '''Fit trainning data.

        Parameter
        -------------
        X: {array-like}, shape = [n_examples, n_features] (n mu/hng, n chiu/ct)
          Trainning vectors, where n_examples is the number of examples 
          and n_features is the number of features.
        y: array-like, shape = [n_examples] (n nhn/im)
          Target values.

        Returns
        ------------
        self: object'''
        rgen = np.random.RandomState(self.random_state)
        self.w_ = rgen.normal(loc = 0.0, scale = 0.01, size = 1+X.shape[1])
        self.cost_ = []

        for i in range(self.n_iter):
            net_input = self.net_input(X)
            output = self.activation(net_input)
            errors = y - output
            self.w_[1:] += self.eta * X.T.dot(errors)
            self.w_[0]  += self.eta * errors.sum()

          # note that we compute the logistic 'cost' now
          # instead of the sum of squared errors cost

            cost = -y.dot(np.log(output)-(1-y).dot(1-np.log(1-output)))
            self.cost_.append(cost)
        return self

    def net_input(self, X):
        '''Calculate net input'''
        return np.dot(X, self.w_[1:]) + self.w_[0]

    def activation(self, z):
        '''Compute logistic sigmoid activation'''
        return 1./(1+np.exp(-np.clip(z, -250, 250)))
  
    def predict(self, X):
        '''Return class label after unit step'''
        return np.where(self.net_input(X) >= 0.0, 1, 0)
        # eqivalent to:
        # return np.where(self.activation(self.net_input(X)) >= 0.5, 1, 0)

V d trong 1 m hnh hc my, ta cn ti u activation function nht c th ng vi tp d liu trainning v da vo phn loi/ d on d liu test. M hnh cng khp (fit) vi d liu th cng tt. Mt m hnh tt l m hnh c tnh tng qut. Regularization l phng php "trng pht" cc im d liu c trng s cao.

Mt phng php regularization ph bin l L2 REGULARIZATION hay l weight decay.

Phng php ny dng iu chnh phc tp ca m hnh bng cch tng hiu sut ca m hnh (gim trng s => gim lch d liu).

$\frac{\lambda}{2}||w||_2^2 = \frac{\lambda}{2} \sum_{j=1}^m {w_j}^2$

$\lambda$ c gi l tham s chnh quy ($\lambda = \frac{1}{c}$) v d, khi thm cng thc trn vo cost function ca Logistic Regression, trng s ca d liu s c gim bt:
$J(w) = - \sum_{i=1}^N(y_ilog(\phi(z_i)) + (1 - y_i)log(1 - \phi(z_i))) + \frac{\lambda}{2}||w||_2^2$
